package main 

import (
	"fmt"

	xssfox "github.com/JGPatelOfficial/xssfox/v2/lib"
)

func main() {
	opt := xssfox.Options{
		Cookie:     "ABCD=1234",
	}
	result, err := xssfox.NewScan(xssfox.Target{
		URL:     "https://xss-game.appspot.com/level1/frame",
		Method:  "GET",
		Options: opt,
	})
	if err != nil {
		fmt.Println(err)
	} else {
		fmt.Println(result)
	}
}
